import matplotlib.pyplot as plt
import pandas as pd

# 데이터 준비
years = [2022, 2023, 2024, 2025, 2026]
revenue = [1854, 1911, 2192, 2597, 2964]  # 매출액
operating_income = [752, 768, 801, 1029, 1204]  # 영업이익
net_income = [500, 595, 740, 840, 969]  # 순이익

# 데이터프레임 생성
data = pd.DataFrame({'년도': years, '매출액': revenue, '영업이익': operating_income, '순이익': net_income})

# 폰트 설정
plt.rc('font', family='NanumGothic')

# 그래프 생성
plt.figure(figsize=(10, 6))
plt.plot(data['년도'], data['매출액'], marker='o', label='매출액 (십억원)', color='blue')
plt.plot(data['년도'], data['영업이익'], marker='o', label='영업이익 (십억원)', color='green')
plt.plot(data['년도'], data['순이익'], marker='o', label='순이익 (십억원)', color='red')

# 제목 및 레이블 설정
plt.title('크래프톤 주요 재무 지표 (2022-2026)')
plt.xlabel('년도')
plt.ylabel('금액 (십억원)')
plt.xticks(years)
plt.yticks()
plt.legend()

# 그래프 저장
plt.savefig('output/Krafton_financials.png', bbox_inches='tight')
plt.close()
