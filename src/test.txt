import matplotlib.pyplot as plt
import pandas as pd

# 데이터 생성
quarters = ['1Q24', '2Q24', '3Q24E', '4Q24E']
revenue = [6659, 7070, 6686, 8359]

# 데이터프레임 생성
data = pd.DataFrame({'Quarter': quarters, 'Revenue (십억 원)': revenue})

# 그래프 생성
plt.figure(figsize=(10, 6))
plt.bar(data['Quarter'], data['Revenue (십억 원)'], color='skyblue')

# 제목 및 레이블 추가
plt.title('크래프톤 2024년 분기별 매출액', fontsize=16, fontfamily='NanumGothic')
plt.xlabel('분기', fontsize=14, fontfamily='NanumGothic')
plt.ylabel('매출액 (십억 원)', fontsize=14, fontfamily='NanumGothic')

# y축 표시 형식 지정
plt.xticks(rotation=45)
plt.grid(axis='y')

# 그래프 저장
plt.savefig('output/krafton_2024_revenue.png')
plt.show()
